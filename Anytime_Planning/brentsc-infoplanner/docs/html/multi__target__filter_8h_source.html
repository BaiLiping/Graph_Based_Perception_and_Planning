61 
62  // Get system state from robot.
63  state x_t = robot.getState();
64  VectorXd mean_prior = robot.tmm->GetTargetState();
65  MatrixXd cov_prior = robot.tmm->GetCovarianceMatrix();
66  // Get problem dimension.
67  int num_targets = robot.tmm->num_targets();
68  int y_dim = robot.tmm->target_dim / num_targets;
69  int z_dim = robot.sensor->z_dim;
70  // Allocate matrices.
71  MatrixXd A(num_targets * y_dim, num_targets * y_dim);
72  MatrixXd W(num_targets * y_dim, num_targets * y_dim);
73  MatrixXd H(num_targets * z_dim, num_targets * y_dim); // Jacobian w.r.t. range bearing for y_dim
74  MatrixXd V(num_targets * z_dim, num_targets * z_dim); // target
75 
76  // Get Target Motion Model, and Sensor Observation Model
77  robot.tmm->GetJacobian(A, W);
78  robot.sensor->GetJacobian(H, V, x_t, robot.tmm);
79 
80  // Compute Innovation
81  VectorXd innovation(z_dim * num_targets);
82  for (int i = 0; i < m.size(); i++) {
83  int ID = m[i].ID;
84  // Get current Robot Belief
85  VectorXd robot_belief = robot.tmm->targets.at(ID)->state;
86  // Compute Innovation.
87  innovation.segment(i * z_dim, z_dim) = m[i].z - robot.sensor->Sense(x_t, robot_belief);
88  }
89  GaussianBelief result = KalmanFilter::KF(innovation, mean_prior, cov_prior, A, W, H, V, debug);
