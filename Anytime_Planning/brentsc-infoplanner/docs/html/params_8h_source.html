104  std::vector<Robot<SE3Pose>> BuildRobots(std::string simulation_yaml) {

112  BuildEnvironment(
113  YAML::LoadFile(simulation_yaml)["worldConfig"].as<std::string>(),

116  // Load robot state
117 
118  std::vector<double> initialState = robot_entry["initialState"].as<std::vector<double>>();

122  SE3Pose state = BuildSE3Pose(initialState);
123  if (debug)

126

130  // Load the sensors

206  std::shared_ptr<InfoTargetModel> BuildInfoTMM(std::string target_yaml) {
207 
208  std::shared_ptr<TargetModel> tmm = BuildTMM(target_yaml); // Parse the target model information.
209  YAML::Node tmm_node = YAML::LoadFile(target_yaml);
210  // Read in YAML Parameters
211  int use_prior = tmm_node["use_prior"].as<int>();
212  int velocity = tmm_node["velocity"].as<int>();
213  int num_targets = tmm->num_targets();
214  int dim = tmm->target_dim / num_targets;
215  double q = tmm_node["q"].as<double>();
216  double cov_pos = tmm_node["cov_pos"].as<double>();
217  double cov_vel = tmm_node["cov_vel"].as<double>();
218  SystemModel m = systemMatrix(q, dim, cov_pos, velocity, cov_vel); // System Dynamics
219 
220  // Build InfoTargetModel
221  std::shared_ptr<InfoTargetModel> model = std::make_shared<InfoTargetModel>(InfoTargetModel());
222  if (use_prior) { // Add target priors!
223  for (const auto &pair : tmm->targets) {
224  int ID = pair.first;
225  auto target = pair.second;
226  model->AddTarget(ID, InfoTarget(ID, target->state, target->A, target->W, m.Sigma));
227  }
228  }

242  double cov_pos = tmm_node["cov_pos"].as<double>();
243  double cov_vel = tmm_node["cov_vel"].as<double>();

246  const std::vector<int> &env_da = tmm_node["env_da"].as<std::vector<int>>();

249  SystemModel m = systemMatrix(q, dim, cov_pos, velocity, cov_vel); // System Dynamics

253  std::vector<int> da = env_da;
254  for (int i = 0; i < da.size(); i++) {

256  VectorXd state = y.segment(dim * i, dim);
257  tmm->AddTarget(ID, Target(ID, state, m.A, m.W));
258  }

380  SystemModel systemMatrix(double q, int dim, double cov_pos, int velocity = 0, double cov_vel = -1) {
381  SystemModel m;
382  if (velocity) {
383  m.A = MatrixXd::Identity(4, 4);
384  m.W = MatrixXd::Identity(4, 4);
385  m.Sigma = MatrixXd::Zero(4, 4);
386  // Set Dynamics matrix
387  m.A.block(0, 2, 2, 2) = samp * MatrixXd::Identity(2, 2);
388  // Set Noise matrix
389  m.W.block(0, 0, 2, 2) = samp * samp * samp / 3 * MatrixXd::Identity(2, 2);
390  m.W.block(0, 2, 2, 2) = samp * samp / 2 * MatrixXd::Identity(2, 2);
391  m.W.block(2, 0, 2, 2) = samp * samp / 2 * MatrixXd::Identity(2, 2);
392  m.W.block(2, 2, 2, 2) = samp * MatrixXd::Identity(2, 2);
393  m.W = m.W * q; // Apply diffusion strength
394  // Set Covariance Matrix
395  m.Sigma.block(0, 0, 2, 2) = cov_pos * MatrixXd::Identity(2, 2);
396  m.Sigma.block(2, 2, 2, 2) = cov_vel * MatrixXd::Identity(2, 2);
397  } else { // Position only, e.g. Random Walk
398  m.A = MatrixXd::Identity(2, 2);
399  m.W = q * MatrixXd::Identity(2, 2);
400  m.Sigma = cov_pos * MatrixXd::Identity(2, 2);
401  }
402  return m;
403  }

405  VectorXd stateMatToVector(MatrixXd y, int velocity, int target_dim) {
406  if (velocity) {
407  // First check if we need to append velocities
408  // If needed, add 0 initial velocity to the target state
409  if (y.cols() == 2) {
410  MatrixXd y_temp(y.rows(), 4);
411  y_temp.setZero();
412  y_temp.block(0, 0, y.rows(), 2) = y;
413  y.resize(y_temp.rows(), y_temp.cols());
414  y = y_temp;
415  }
416  }
417  int n_targets = y.rows();
418  // Now construct the vector.
419  VectorXd output(target_dim * n_targets);
420  output.fill(0.0);
421  // Append all targets
422  for (int i = 0; i < n_targets; i++) {
423  output.segment(i * target_dim, target_dim) = y.row(i);
424  }
425  return output;
426  }
